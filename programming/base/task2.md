## Приклад програми
Розрахунок температури в фаренгейтах
Формула F=(5/9×C)+32

    Program HelloWorld(output);
    var c, f: integer;
    begin
      writeln('Яка температура в цельсіях?');
      read(c);
      f := c * 5 div 9 + 32;
      writeln('Температура в фаренгейтах: ', f);
    end.
